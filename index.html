function dibujarGrafico(ancho, alto, factor, porcentaje) { let canvas = document.getElementById("grafico") let c = canvas.getContext("2d") c.clearRect(0,0,canvas.width,canvas.height) let escala = Math.min(canvas.width/ancho, canvas.height/alto) let w = ancho * escala let h = alto * escala let origenX = 20 let origenY = 20 // Marco escalerilla c.strokeRect(origenX, origenY, w, h) // Zona permitida NEC let anchoPermitido = w * factor c.fillStyle = "rgba(255,165,0,0.3)" c.fillRect(origenX, origenY, anchoPermitido, h) // Posición inicial → esquina inferior izquierda let x = origenX let yBase = origenY + h let altoColumna = 0 let anchoColumna = 0 cables.forEach(cable => { let r = (cable.d/2) * escala let diam = r * 2 for (let i = 0; i < cable.n; i++) { // Si no cabe verticalmente → nueva columna a la derecha if (altoColumna + diam > h) { x += anchoColumna altoColumna = 0 anchoColumna = 0 } let y = yBase - altoColumna - r c.beginPath() c.arc(x + r, y, r, 0, Math.PI*2) c.stroke() altoColumna += diam anchoColumna = Math.max(anchoColumna, diam) } }) }
